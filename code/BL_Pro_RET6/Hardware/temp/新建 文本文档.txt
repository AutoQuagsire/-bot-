    ' -- 新增：辅助圆角路径生成方法 --
    Private Shared Function GetRoundedPath(rect As Rectangle, radius As Integer) As GraphicsPath
        Dim path As New GraphicsPath()
        path.AddArc(rect.X, rect.Y, radius, radius, 180, 90)
        path.AddArc(rect.Right - radius, rect.Y, radius, radius, 270, 90)
        path.AddArc(rect.Right - radius, rect.Bottom - radius, radius, radius, 0, 90)
        path.AddArc(rect.X, rect.Bottom - radius, radius, radius, 90, 90)
        path.CloseFigure()
        Return path
    End Function

    ' -- 新增：为控件裁剪圆角区域 --
    Private Sub SetRoundedRegion(ctrl As Control, radius As Integer)
        ctrl.Region = New Region(GetRoundedPath(New Rectangle(0, 0, ctrl.Width, ctrl.Height), radius))
    End Sub

    ' -- 替换原 AddBorderPaint，改用平滑圆角底色背景 --
    Private Shared Sub AddRoundedBackground(pnl As Panel, bgColor As Color, radius As Integer)

            e.Graphics.SmoothingMode = SmoothingMode.AntiAlias
            Dim path As GraphicsPath = GetRoundedPath(New Rectangle(0, 0, pnl.Width - 1, pnl.Height - 1), radius)
            Using brush As New SolidBrush(bgColor)
                e.Graphics.FillPath(brush, path)
            End Using
        End Sub
    End Sub
